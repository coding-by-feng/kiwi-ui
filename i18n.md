## 3. Create i18n Setup File

### src/i18n/index.js
```javascript
import Vue from 'vue'
import VueI18n from 'vue-i18n'
import { getStore, setStore } from '@/util/store'

// Import language files
import enLocale from '@/locales/en'
import zhLocale from '@/locales/zh'
import jaLocale from '@/locales/ja'
import frLocale from '@/locales/fr'
import esLocale from '@/locales/es'

// Import Element UI language packs
import enElementLocale from 'element-ui/lib/locale/lang/en'
import zhElementLocale from 'element-ui/lib/locale/lang/zh-CN'
import jaElementLocale from 'element-ui/lib/locale/lang/ja'
import frElementLocale from 'element-ui/lib/locale/lang/fr'
import esElementLocale from 'element-ui/lib/locale/lang/es'

Vue.use(VueI18n)

// Merge application and Element UI messages
const messages = {
  en: {
    ...enLocale,
    ...enElementLocale
  },
  zh: {
    ...zhLocale,
    ...zhElementLocale
  },
  ja: {
    ...jaLocale,
    ...jaElementLocale
  },
  fr: {
    ...frLocale,
    ...frElementLocale
  },
  es: {
    ...esLocale,
    ...esElementLocale
  }
}

// Get saved language or detect from browser
function getDefaultLanguage() {
  const savedLanguage = getStore({ name: 'app_language' })
  if (savedLanguage && messages[savedLanguage]) {
    return savedLanguage
  }

  // Detect browser language
  const browserLanguage = navigator.language || navigator.userLanguage
  
  if (browserLanguage.startsWith('zh')) {
    return 'zh'
  } else if (browserLanguage.startsWith('ja')) {
    return 'ja'
  } else if (browserLanguage.startsWith('fr')) {
    return 'fr'
  } else if (browserLanguage.startsWith('es')) {
    return 'es'
  }
  
  return 'en' // Default to English
}

const i18n = new VueI18n({
  locale: getDefaultLanguage(),
  fallbackLocale: 'en',
  messages,
  silentTranslationWarn: true,
  silentFallbackWarn: true
})

// Save language preference when changed
export function setLanguage(lang) {
  i18n.locale = lang
  setStore({
    name: 'app_language',
    content: lang,
    type: 'local'
  })
  
  // Update HTML lang attribute for accessibility
  document.documentElement.lang = lang
}

// Get current language
export function getCurrentLanguage() {
  return i18n.locale
}

// Get available languages
export function getAvailableLanguages() {
  return [
    { code: 'en', name: 'English', nativeName: 'English' },
    { code: 'zh', name: 'Chinese', nativeName: '中文' },
    { code: 'ja', name: 'Japanese', nativeName: '日本語' },
    { code: 'fr', name: 'French', nativeName: 'Français' },
    { code: 'es', name: 'Spanish', nativeName: 'Español' }
  ]
}

export default i18n
```

```javascript
/**
 * Language detection utilities for auto-switching based on content

// Common Chinese characters detection
export function isChineseText(text) {
    if (!text) return false
    const chineseRegex = /[\u4e00-\u9fa5]/
    return chineseRegex.test(text)
}

// Common Japanese characters detection
export function isJapaneseText(text) {
    if (!text) return false
    const japaneseRegex = /[\u3040-\u309f\u30a0-\u30ff\u4e00-\u9faf]/
    return japaneseRegex.test(text)
}

// French text detection (basic)
export function isFrenchText(text) {
    if (!text) return false
    const frenchWords = ['le', 'la', 'les', 'de', 'du', 'des', 'et', 'est', 'une', 'un', 'dans', 'pour', 'avec', 'sur', 'être', 'avoir']
    const words = text.toLowerCase().split(/\s+/)
    const frenchWordCount = words.filter(word => frenchWords.includes(word)).length
    return frenchWordCount > words.length * 0.15
}

// Spanish text detection (basic)
export function isSpanishText(text) {
    if (!text) return false
    const spanishWords = ['el', 'la', 'los', 'las', 'de', 'del', 'y', 'es', 'una', 'un', 'en', 'para', 'con', 'por', 'ser', 'tener']
    const words = text.toLowerCase().split(/\s+/)
    const spanishWordCount = words.filter(word => spanishWords.includes(word)).length
    return spanishWordCount > words.length * 0.15
}

// Auto-detect language based on text content
export function detectLanguage(text) {
    if (!text || text.trim().length === 0) return 'en'
    
    if (isChineseText(text)) return 'zh'
    if (isJapaneseText(text)) return 'ja'
    if (isFrenchText(text)) return 'fr'
    if (isSpanishText(text)) return 'es'
    
    return 'en' // Default to English
}

// Get appropriate target language for translation based on detected source
export function getTranslationTarget(sourceText, currentLang) {
    const detectedLang = detectLanguage(sourceText)
    
    // If detected language is different from current UI language, use current UI language as target
    if (detectedLang !== currentLang) {
        return currentLang
    }
    
    // If same language, translate to English as fallback
    return 'en'
}

export default {
    isChineseText,
    isJapaneseText,
    isFrenchText,
    isSpanishText,
    detectLanguage,
    getTranslationTarget
}
```

```javascript
/**
 * Migration script to help identify hardcoded strings that need i18n
 * Run with: node scripts/migrate-i18n.js
 */

const fs = require('fs')
const path = require('path')

// Directories to scan
const srcDir = path.join(__dirname, '../src')

// Common Chinese characters and patterns
const chinesePattern = /[\u4e00-\u9fa5]+/g
const hardcodedStringPattern = /(["'])((?:[^"'\\]|\\.)*)(\1)/g

function scanFile(filePath) {
    const content = fs.readFileSync(filePath, 'utf8')
    const lines = content.split('\n')
    const issues = []

    lines.forEach((line, index) => {
        // Find Chinese text
        const chineseMatches = line.match(chinesePattern)
        if (chineseMatches) {
            issues.push({
                file: filePath,
                line: index + 1,
                type: 'chinese',
                text: chineseMatches.join(', '),
                content: line.trim()
            })
        }

        // Find potential hardcoded English strings in templates
        if (filePath.endsWith('.vue') && line.includes('<') && !line.includes('$t(')) {
            const stringMatches = line.match(hardcodedStringPattern)
            if (stringMatches) {
                stringMatches.forEach(match => {
                    const text = match.slice(1, -1) // Remove quotes
                    if (text.length > 3 && /^[a-zA-Z\s]+$/.test(text) && !text.includes('el-icon')) {
                        issues.push({
                            file: filePath,
                            line: index + 1,
                            type: 'english',
                            text: text,
                            content: line.trim()
                        })
                    }
                })
            }
        }
    })

    return issues
}

function scanDirectory(dir) {
    const items = fs.readdirSync(dir)
    let allIssues = []

    items.forEach(item => {
        const itemPath = path.join(dir, item)
        const stat = fs.statSync(itemPath)

        if (stat.isDirectory() && !item.startsWith('.') && item !== 'node_modules') {
            allIssues = allIssues.concat(scanDirectory(itemPath))
        } else if (stat.isFile() && (item.endsWith('.vue') || item.endsWith('.js'))) {
            allIssues = allIssues.concat(scanFile(itemPath))
        }
    })

    return allIssues
}

console.log('Scanning for hardcoded strings that need i18n...')
const issues = scanDirectory(srcDir)

// Group by file
const groupedIssues = {}
issues.forEach(issue => {
    if (!groupedIssues[issue.file]) {
        groupedIssues[issue.file] = []
    }
    groupedIssues[issue.file].push(issue)
})

// Output results
console.log(`\nFound ${issues.length} potential i18n issues in ${Object.keys(groupedIssues).length} files:\n`)

Object.keys(groupedIssues).forEach(file => {
    console.log(`📁 ${file.replace(srcDir, '')}`)
    groupedIssues[file].forEach(issue => {
        console.log(`  Line ${issue.line}: [${issue.type.toUpperCase()}] ${issue.text}`)
        console.log(`    ${issue.content}`)
    })
    console.log('')
})

// Generate summary
const chineseIssues = issues.filter(i => i.type === 'chinese')
const englishIssues = issues.filter(i => i.type === 'english')

console.log(`\n📊 Summary:`)
console.log(`   Chinese text: ${chineseIssues.length} instances`)
console.log(`   English text: ${englishIssues.length} instances`)
console.log(`   Total: ${issues.length} instances`)
```

### src/mixins/i18nMixin.js
```javascript
import { getCurrentLanguage, setLanguage } from '@/i18n'
import { detectLanguage } from '@/util/langDetect'

export default {
  computed: {
    currentLanguage() {
      return getCurrentLanguage()
    },
    
    isRTL() {
      // Add RTL language support if needed
      const rtlLanguages = ['ar', 'he', 'fa']
      return rtlLanguages.includes(this.currentLanguage)
    }
  },
  
  methods: {
    // Translate with fallback
    $tt(key, params = {}, fallback = '') {
      if (this.$te(key)) {
        return this.$t(key, params)
      }
      return fallback || key
    },
    
    // Translate with pluralization
    $tp(key, count, params = {}) {
      return this.$tc(key, count, { count, ...params })
    },
    
    // Format time with localization
    formatTime(timestamp) {
      if (!timestamp) return ''
      
      const date = new Date(timestamp)
      const now = new Date()
      const diffTime = now - date
      const diffDays = Math.floor(diffTime / (1000 * 60 * 60 * 24))
      
      if (diffDays === 0) {
        return this.$t('time.today') + ' ' + date.toLocaleTimeString(this.currentLanguage)
      } else if (diffDays === 1) {
        return this.$t('time.yesterday') + ' ' + date.toLocaleTimeString(this.currentLanguage)
      } else if (diffDays <= 7) {
        return this.$t('time.daysAgo', { days: diffDays })
      } else {
        return date.toLocaleDateString(this.currentLanguage)
      }
    },
    
    // Auto-detect and suggest language
    suggestLanguageForContent(text) {
      const detected = detectLanguage(text)
      if (detected !== this.currentLanguage) {
        return {
          suggested: detected,
          confidence: this.getDetectionConfidence(text, detected)
        }
      }
      return null
    },
    
    getDetectionConfidence(text, detectedLang) {
      // Simple confidence scoring
      if (detectedLang === 'zh' && /[\u4e00-\u9fa5]/.test(text)) {
        const chineseRatio = (text.match(/[\u4e00-\u9fa5]/g) || []).length / text.length
        return Math.min(chineseRatio * 2, 1)
      }
      return 0.5 // Default confidence
    },
    
    // Change language with confirmation
    async changeLanguageWithConfirm(newLang) {
      try {
        await this.$confirm(
          this.$t('messages.changeLanguageConfirm'),
          this.$t('ui.confirm'),
          {
            confirmButtonText: this.$t('ui.confirm'),
            cancelButtonText: this.$t('ui.cancel'),
            type: 'info'
          }
        )
        
        setLanguage(newLang)
        this.$message.success(this.$t('messages.languageChanged'))
        
        // Reload to apply changes
        setTimeout(() => {
          window.location.reload()
        }, 500)
        
      } catch {
        // User cancelled
      }
    }
  }
}
```
